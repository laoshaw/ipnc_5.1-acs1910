    17    /*!
    18     *  ======== C64P ========
    19     *  TI C64P little endian target -- for ELF
    20     */
    21    metaonly module C64P inherits ITarget {
    22        override readonly config string name                 = "C64P";
    23        override readonly config string suffix               = "e64P";
    24        override readonly config string isa                  = "64P";
    25        override readonly config xdc.bld.ITarget.Model model = {
    26            endian: "little",
    27            shortEnums: true
    28        };
    29        override readonly config xdc.bld.ITarget.Module base = ti.targets.C62;
    30    
    31        /*
    32         *  ======== compatibleSuffixes ========
    33         */
    34        override config String compatibleSuffixes[] = ["", "e64", "e62"];
    35    
    36        override readonly config string rts = "ti.targets.rts6000";
    37        override config string platform     = "ti.platforms.sim6xxx:TMS320CDM420";
    38        
    39        override readonly config ti.targets.ITarget.Command ar = {
    40            cmd: "ar6x",
    41            opts: "rq"
    42        };
    43    
    44        override readonly config ti.targets.ITarget.Command cc = {
    45            cmd: "cl6x -c",
    46            opts: "-mv64p --abi=elfabi"
    47        };
    48    
    49        override readonly config ti.targets.ITarget.Command vers = {
    50            cmd: "cl6x",
    51            opts: "-version"
    52        };
    53    
    54        override readonly config ti.targets.ITarget.Command asm = {
    55            cmd: "cl6x -c",
    56            opts: "-mv64P --abi=elfabi"
    57        };
    58    
    59        override readonly config ti.targets.ITarget.Command lnk = {
    60            cmd: "lnk6x",
    61            opts: "--abi=elfabi"
    62        };
    63        
    64        /*!
    65         *  ======== asmOpts ========
    66         *  User configurable assembler options.
    67         *
    68         *  Defaults:
    69         *  @p(dlist)
    70         *      -`-qq`
    71         *          super quiet mode
    72         */
    73        override config ti.targets.ITarget.Options asmOpts = {
    74            prefix: "-qq",
    75            suffix: ""
    76        };
    77    
    78        /*!
    79         *  ======== ccOpts ========
    80         *  User configurable compiler options.
    81         *
    82         *  Defaults:
    83         *  @p(dlist)
    84         *      -`-qq`
    85         *          super quiet mode
    86         *      -`-pdsw225`
    87         *          generate a warning for implicitly declared functions; i.e.,
    88         *          functions without prototypes
    89         */
    90        override config ti.targets.ITarget.Options ccOpts = {
    91            prefix: "-qq -pdsw225",
    92            suffix: ""
    93        };
    94    
    95        /*!
    96         *  ======== ccConfigOpts ========
    97         *  User configurable compiler options for the generated config C file.
    98         *
    99         *  -mo places all functions into subsections
   100         *  --no_compress helps with compile time with no real difference in
   101         *  code size since the generated config.c is mostly data and small
   102         *  function stubs.
   103         */
   104        override config ti.targets.ITarget.Options ccConfigOpts = {
   105            prefix: "$(ccOpts.prefix) -mo --no_compress",
   106            suffix: "$(ccOpts.suffix)"
   107        };
   108    
   109        /*!
   110         *  ======== lnkOpts ========
   111         *  User configurable linker options.
   112         *
   113         *  Defaults:
   114         *  @p(dlist)
   115         *      -`-w`
   116         *          Display linker warnings
   117         *      -`-q`
   118         *          Quite run
   119         *      -`-u`
   120         *          Place unresolved external symbol into symbol table
   121         *      -`-c`
   122         *          ROM autoinitialization model
   123         *      -`-m`
   124         *          create a map file
   125         *      -`-l`
   126         *          archive library file as linker input
   127         */
   128        override config ti.targets.ITarget.Options lnkOpts = {
   129            prefix: "-w -q -u _c_int00",
   130            suffix: "-c -m $(XDCCFGDIR)/$@.map -l $(rootDir)/lib/rts64plus_elf.lib"
   131        };
   132            
   133        override config string includeOpts = "-I$(rootDir)/include";
   134    
   135        override readonly config Int bitsPerChar = 
   136            ti.targets.C62.bitsPerChar;
   137    }
