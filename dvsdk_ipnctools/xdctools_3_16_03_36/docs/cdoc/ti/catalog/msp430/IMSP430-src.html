    18    /*!
    19     *  ======== IMSP430 ========
    20     *  An interface implemented by all MSP430 devices.
    21     *
    22     *  This interface is defined to factor common data about this family into
    23     *  a single place; all MSP430 devices extend this interface.
    24     */
    25    metaonly interface IMSP430 inherits ti.catalog.ICpuDataSheet
    26    {
    27    instance:
    28        override config string   cpuCore        = "MSP430";
    29        override config string   isa            = "430";
    30        override config int     minProgUnitSize = 1;
    31        override config int     minDataUnitSize = 1;    
    32        override config int     dataWordSize    = 2;
    33    };
